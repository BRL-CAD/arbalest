void MainWindow::objectTreeWidgetSelectionChanged(int fullPath) {
    //documents[activeDocumentId]->getProperties()->bindObject(documents[activeDocumentId]);
    documents[activeDocumentId]->getObjectTree()->changeSubTreeVisibility(0, false);
    documents[activeDocumentId]->getObjectTree()->changeSubTreeVisibility(fullPath, true);
    documents[activeDocumentId]->getObjectTree()->rebuildVisibleDisplayListIds();
    documents[activeDocumentId]->getDisplay()->refresh();
}
